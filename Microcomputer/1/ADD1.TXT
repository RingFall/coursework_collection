DATA SEGMENT
DATA1 DB '8'
DATA2 DB '3'
RES DB 1 DUP(0)
DATA ENDS
CODE SEGMENT
ASSUME CS:CODE,DS:DATA
START: MOV AX,DATA
MOV DS,AX
MOV CX,1
MOV SI,0
CLC
CYCLE: MOV AL,DATA1[SI]
 ADC AL,DATA2[SI]
 AAA
MOV RES[SI],AL
INC SI
LOOP CYCLE
JNC NOCY
MOV AH,2
MOV DL,31H
INT 21H
NOCY: MOV SI,1
DISP: ADD RES[SI-1],30H
 MOV DL,RES[SI-1]
 MOV AH,02
INT 21H
DEC SI
JNZ DISP
MOV AH,4CH
INT 21H
CODES ENDS
END START
